1. Basic Implementation

// Function to track citation clicks
async function trackCitationClick(citationData) {
    const endpoint = '/api/TrackCitationClick';
    
    try {
        const response = await fetch(endpoint, {
            method: 'POST',
            headers: {
                'Content-Type': 'application/json'
            },
            body: JSON.stringify({
                conversationId: citationData.conversationId,
                citationTitle: citationData.title,
                citationUrl: citationData.url,
                userId: citationData.userId || 'anonymous',
                timestamp: new Date().toISOString()
            })
        });
        
        if (response.ok) {
            console.log('Citation click tracked successfully');
        }
    } catch (error) {
        console.error('Failed to track citation:', error);
        // Don't block user navigation on tracking failure
    }
}

// Add click handler to citation links
document.querySelectorAll('.citation-link').forEach(link => {

    });
});

2. React Component Example

import React from 'react';

const CitationLink = ({ citation, conversationId, userId }) => {
    const handleClick = async (e) => {
        e.preventDefault();
        
        // Track the click
        try {
            await fetch('/api/TrackCitationClick', {
                method: 'POST',
                headers: { 'Content-Type': 'application/json' },
                body: JSON.stringify({
                    conversationId,
                    citationTitle: citation.title,
                    citationUrl: citation.url,
                    userId: userId || 'anonymous',
                    timestamp: new Date().toISOString()
                })
            });
        } catch (error) {
            console.error('Citation tracking failed:', error);
        }
        
        // Open link
        window.open(citation.url, '_blank');
    };
    
    return (
        <a 
            href={citation.url}
            onClick={handleClick}
            className="citation-link"
            target="_blank"
            rel="noopener noreferrer"
        >
            {citation.title}
        </a>
    );
};

export default CitationLink;
